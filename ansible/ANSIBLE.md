### Using Existing Ansible Role:

```
---
- hosts: all
  become: true
  roles:
    - geerlingguy.docker
    # - roles/docker
  post_tasks:
    # below we check results of a role execution
    - name: Verify docker installation
      command: docker --version
      register: docker_version

    - name: Print docker version
      debug:
        msg: "{{ docker_version.stdout }}"
```
